<script>
function checkResult() {

  let box = document.getElementById("resultBox");

  // LOADING EFFECT
  box.style.backgroundColor = "gray";
  box.innerText = "⏳ Analyzing conditions...";

  setTimeout(() => {

    let mode = document.getElementById("mode").value;
    let seed = document.getElementById("seed").value;
    let soil = document.getElementById("soil").value;
    let temp = document.getElementById("temp").value;
    let water = document.getElementById("water").value;
    let experience = document.getElementById("experience").value;

    let score = 0;
    let result = "";
    let color = "";

    if (seed === "Local") score += 2;
    if (soil === "Black Soil") score += 2;
    if (temp === "Medium") score += 2;
    if (water === "Medium") score += 2;
    if (experience === "Experienced") score += 2;

    if (seed === "Hybrid") score = 0;

    if (score >= 7) {
      result = "✅ Reusable\n";
      color = "#00c853";
    }
    else if (score >= 4) {
      result = "⚠️ Risky\n";
      color = "#ff9800";
    }
    else {
      result = "❌ Not Recommended\n";
      color = "#d32f2f";
    }

    result += "Score: " + score + "/10\n";

    if (seed === "Hybrid") {
      result += "Reason: Hybrid seeds cannot be reused.\n";
    } else {
      result += "Reason: Based on environmental conditions.\n";
    }

    result += "Estimated Saving: ₹1000–₹2000\n";
    result += "Access Mode: " + mode;

    box.innerText = result;
    box.style.backgroundColor = color;

  }, 1200);

}
</script>
